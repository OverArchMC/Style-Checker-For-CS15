// BAD: line over 80 characters

// BAD: while (true) and while (false)
void infiniteLoops() {
    while (true) {
        std::cout << "infinite" << std::endl;
        break;
    }
    while (false) {
        std::cout << "never runs" << std::endl;
    }
}

// BAD: break statements
void hasBreak() {
    for (int i = 0; i < 10; i++) {
        if (i == 5) {
            break;
        }
    }
}

// BAD: missing spaces before parens
void badSpacing() {
    for(int i = 0; i < 10; i++) {
        if(i % 2 == 0) {
            std::cout << i << std::endl;
        }
        while(false) {
            std::cout << "never" << std::endl;
        }
    }
}

// BAD: bad indentation
void badIndentation() {
        int x = 5;

    }
}

// BAD: single line if/for/while
void singleLineStatements() {
    int x = 5;
    if (x > 3) std::cout << x << std::endl;
    for (int i = 0; i < 5; i++) std::cout << i << std::endl;
    while (x > 0) x--;
}

// GOOD: no violations
void cleanFunction(int x) {
    if (x > 0) {
        std::cout << x << std::endl;
    }
    for (int i = 0; i < x; i++) {
        std::cout << i << std::endl;
    }
}
